34 template<typename TChainInput, typename TGroupIdentifierFn, typename TGroupIdent>
35 struct IteratorTrait<op::GroupBy<TChainInput, TGroupIdentifierFn, TGroupIdent>> {
36 using ChainInputIterator = IteratorTrait<TChainInput>;
37 using OwnedInputItem = typename TChainInput::ItemOwned;
38 // CXXIter Interface
39 using Self = op::GroupBy<TChainInput, TGroupIdentifierFn, TGroupIdent>;
40 using Item = std::pair<const TGroupIdent, std::vector<OwnedInputItem>>;
41
42 static inline IterValue<Item> next(Self& self) {
43 // we have to drain the input in order to be able to calculate the groups
44 // so we do that on the first invocation, and then yield from the calculated result.
45 if(!self.groupCache.has_value()) [[unlikely]] {
46 std::unordered_map<TGroupIdent, std::vector<OwnedInputItem>> groupCache;
47 while(true) {
48 auto item = ChainInputIterator::next(self.input);
49 if(!item.has_value()) [[unlikely]] { break; } // group cache building complete
50 TGroupIdent itemGroup = self.groupIdentFn(item.value());
51 if(groupCache.contains(itemGroup)) {
52 groupCache[itemGroup].push_back(item.value());
53 } else {
54 groupCache[itemGroup] = { item.value() };
55 }
56 }
57 self.groupCache.emplace(std::move(groupCache));
58 }
59
60 using GroupCacheIterator = IteratorTrait<typename Self::GroupCache>;
61 typename Self::GroupCache& groupedItems = self.groupCache.value();
62 return GroupCacheIterator::next(groupedItems);
63 }
64 static inline SizeHint sizeHint(const Self& self) {
65 SizeHint input = ChainInputIterator::sizeHint(self.input);
66 return SizeHint(1, input.upperBound);
67 }
68 };
