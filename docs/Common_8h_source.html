134 struct SizeHint {
135 constexpr static size_t INFINITE = std::numeric_limits<size_t>::max();
136
137 size_t lowerBound;
138 std::optional<size_t> upperBound;
139
140 size_t expectedResultSize(size_t min = 0) const { return std::min(min, upperBound.value_or(lowerBound)); }
141
142 SizeHint(size_t lowerBound = 0, std::optional<size_t> upperBound = {}) : lowerBound(lowerBound), upperBound(upperBound) {}
143
144 static std::optional<size_t> upperBoundMax(std::optional<size_t> upperBound1, std::optional<size_t> upperBound2) {
145 if(!upperBound1.has_value() || !upperBound2.has_value()) { return {}; } // no upperbound is like Infinity -> higher
146 return std::max(upperBound1.value(), upperBound2.value());
147 }
148 static std::optional<size_t> upperBoundMin(std::optional<size_t> upperBound1, std::optional<size_t> upperBound2) {
149 if(!upperBound1.has_value()) { return upperBound2; }
150 if(!upperBound2.has_value()) { return upperBound1; }
151 return std::min(upperBound1.value(), upperBound2.value());
152 }
153 void subtract(size_t cnt) {
154 lowerBound = (lowerBound > cnt) ? (lowerBound - cnt) : 0;
155 if(upperBound) {
156 upperBound = (upperBound.value() > cnt) ? (upperBound.value() - cnt) : 0;
157 }
158 }
159 void add(const SizeHint& o) {
160 lowerBound += o.lowerBound;
161 if(upperBound.has_value() && o.upperBound.has_value()) {
162 upperBound = upperBound.value() + o.upperBound.value();
163 } else {
164 upperBound = {};
165 }
166 }
167 };
