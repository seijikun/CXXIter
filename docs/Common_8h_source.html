118 struct SizeHint {
119 constexpr static size_t INFINITE = std::numeric_limits<size_t>::max();
120
121 size_t lowerBound;
122 std::optional<size_t> upperBound;
123
124 size_t expectedResultSize(size_t min = 0) const { return std::min(min, upperBound.value_or(lowerBound)); }
125
126 SizeHint(size_t lowerBound = 0, std::optional<size_t> upperBound = {}) : lowerBound(lowerBound), upperBound(upperBound) {}
127
128 static std::optional<size_t> upperBoundMax(std::optional<size_t> upperBound1, std::optional<size_t> upperBound2) {
129 if(!upperBound1.has_value() || !upperBound2.has_value()) { return {}; } // no upperbound is like Infinity -> higher
130 return std::max(upperBound1.value(), upperBound2.value());
131 }
132 static std::optional<size_t> upperBoundMin(std::optional<size_t> upperBound1, std::optional<size_t> upperBound2) {
133 if(!upperBound1.has_value()) { return upperBound2; }
134 if(!upperBound2.has_value()) { return upperBound1; }
135 return std::min(upperBound1.value(), upperBound2.value());
136 }
137 void subtract(size_t cnt) {
138 lowerBound = (lowerBound > cnt) ? (lowerBound - cnt) : 0;
139 if(upperBound) {
140 upperBound = (upperBound.value() > cnt) ? (upperBound.value() - cnt) : 0;
141 }
142 }
143 void add(const SizeHint& o) {
144 lowerBound += o.lowerBound;
145 if(upperBound.has_value() && o.upperBound.has_value()) {
146 upperBound = upperBound.value() + o.upperBound.value();
147 } else {
148 upperBound = {};
149 }
150 }
151 };
