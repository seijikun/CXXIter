51 static void initReverseCache(Self& self) {
52 // drain input iterator into reverse cache
53 typename Self::ReverseCacheContainer reverseCache;
54 reverseCache.reserve(self.input.sizeHint().expectedResultSize());
55 while(true) {
56 auto item = ChainInputIterator::next(self.input);
57 if(!item.has_value()) { break; }
58 reverseCache.push_back(std::move(item.value()));
59 }
60 self.reverseCache = typename Self::ReverseCache(std::move(reverseCache));
61 }
