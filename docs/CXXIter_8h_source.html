705 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
706 std::optional<TResult> mean() {
707 size_t cnt = 0;
708 TResult result = fold(TResult(), [&cnt](TResult& res, Item&& item) {
709 cnt += 1;
710 res += item;
711 });
712 if(cnt > 0) {
713 if constexpr(NORM == StatisticNormalization::N) {
714 return result / static_cast<TCount>(cnt);
715 } else {
716 return result / static_cast<TCount>(cnt - 1);
717 }
718 }
719 return {};
720 }

759 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
760 std::optional<TResult> variance() {
761 TResult sumSquare = TResult();
762 TResult sum = TResult();
763 size_t cnt = 0;
764 forEach([&sumSquare, &sum, &cnt](Item&& item) {
765 sum += item;
766 sumSquare += (item * item);
767 cnt += 1;
768 });
769 if(cnt >= 2) {
770 if constexpr(NORM == StatisticNormalization::N) {
771 TResult E1 = (sumSquare / static_cast<TCount>(cnt));
772 TResult E2 = (sum / static_cast<TCount>(cnt));
773 return E1 - (E2 * E2);
774 } else {
775 TResult E1 = (sum * sum / static_cast<TCount>(cnt));
776 return (sumSquare - E1) / static_cast<TCount>(cnt - 1);
777 }
778 }
779 return {};
780 }
