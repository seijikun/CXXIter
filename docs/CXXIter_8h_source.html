744 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
745 std::optional<TResult> mean(TResult sumStart = TResult()) {
746 size_t cnt = 0;
747 TResult result = fold(sumStart, [&cnt](TResult& res, Item&& item) {
748 cnt += 1;
749 res += item;
750 });
751 if(cnt > 0) {

798 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
799 std::optional<TResult> variance() {
800 TResult sumSquare = TResult();
801 TResult sum = TResult();
802 size_t cnt = 0;
803 forEach([&sumSquare, &sum, &cnt](Item&& item) {
804 sum += item;
805 sumSquare += (item * item);
806 cnt += 1;
807 });
808 if(cnt >= 2) {
809 if constexpr(NORM == StatisticNormalization::N) {
810 TResult E1 = (sumSquare / static_cast<TCount>(cnt));
811 TResult E2 = (sum / static_cast<TCount>(cnt));
812 return E1 - (E2 * E2);
813 } else {
814 TResult E1 = (sum * sum / static_cast<TCount>(cnt));
815 return (sumSquare - E1) / static_cast<TCount>(cnt - 1);
816 }
817 }
818 return {};
819 }
