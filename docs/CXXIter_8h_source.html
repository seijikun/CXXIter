706 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
707 std::optional<TResult> mean() {
708 size_t cnt = 0;
709 TResult result = fold(TResult(), [&cnt](TResult& res, Item&& item) {
710 cnt += 1;
711 res += item;
712 });
713 if(cnt > 0) {
714 if constexpr(NORM == StatisticNormalization::N) {
715 return result / static_cast<TCount>(cnt);
716 } else {
717 return result / static_cast<TCount>(cnt - 1);
718 }
719 }
720 return {};
721 }

760 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
761 std::optional<TResult> variance() {
762 TResult sumSquare = TResult();
763 TResult sum = TResult();
764 size_t cnt = 0;
765 forEach([&sumSquare, &sum, &cnt](Item&& item) {
766 sum += item;
767 sumSquare += (item * item);
768 cnt += 1;
769 });
770 if(cnt >= 2) {
771 if constexpr(NORM == StatisticNormalization::N) {
772 TResult E1 = (sumSquare / static_cast<TCount>(cnt));
773 TResult E2 = (sum / static_cast<TCount>(cnt));
774 return E1 - (E2 * E2);
775 } else {
776 TResult E1 = (sum * sum / static_cast<TCount>(cnt));
777 return (sumSquare - E1) / static_cast<TCount>(cnt - 1);
778 }
779 }
780 return {};
781 }
