704 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
705 std::optional<TResult> mean() {
706 size_t cnt = 0;
707 TResult result = fold(TResult(), [&cnt](TResult& res, Item&& item) {
708 cnt += 1;
709 res += item;
710 });
711 if(cnt > 0) {
712 if constexpr(NORM == StatisticNormalization::N) {
713 return result / static_cast<TCount>(cnt);
714 } else {
715 return result / static_cast<TCount>(cnt - 1);
716 }
717 }
718 return {};
719 }

751 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
752 std::optional<TResult> stddev() {
753 TResult sumSquare = TResult();
754 TResult sum = TResult();
755 size_t cnt = 0;
756 forEach([&sumSquare, &sum, &cnt](Item&& item) {
757 sum += item;
758 sumSquare += (item * item);
759 cnt += 1;
760 });
761 if(cnt >= 2) {
762 if constexpr(NORM == StatisticNormalization::N) {
763 TResult E1 = (sumSquare / static_cast<TCount>(cnt));
764 TResult E2 = (sum / static_cast<TCount>(cnt));
765 return std::sqrt(E1 - (E2 * E2));
766 } else {
767 return std::sqrt(
768 (sumSquare - (sum * sum / static_cast<TCount>(cnt))) / static_cast<TCount>(cnt - 1)
769 );
770 }
771 }
772 return {};
773 }
