786 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
787 std::optional<TResult> mean(TResult sumStart = TResult()) {
788 size_t cnt = 0;
789 TResult result = fold(sumStart, [&cnt](TResult& res, Item&& item) {
790 cnt += 1;
791 res += item;
792 });
793 if(cnt > 0) {
794 if constexpr(NORM == StatisticNormalization::N) {
795 return result / static_cast<TCount>(cnt);
796 } else {
797 return result / static_cast<TCount>(cnt - 1);
798 }
799 }
800 return {};

840 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
841 std::optional<TResult> variance() {
842 TResult sumSquare = TResult();
843 TResult sum = TResult();
844 size_t cnt = 0;
845 forEach([&sumSquare, &sum, &cnt](Item&& item) {
846 sum += item;
847 sumSquare += (item * item);
848 cnt += 1;
849 });
850 if(cnt >= 2) {
851 if constexpr(NORM == StatisticNormalization::N) {
852 TResult E1 = (sumSquare / static_cast<TCount>(cnt));
853 TResult E2 = (sum / static_cast<TCount>(cnt));
854 return E1 - (E2 * E2);
855 } else {
856 TResult E1 = (sum * sum / static_cast<TCount>(cnt));
857 return (sumSquare - E1) / static_cast<TCount>(cnt - 1);
858 }
859 }
860 return {};
861 }
