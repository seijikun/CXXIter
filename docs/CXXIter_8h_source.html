806 template<typename TMinValueExtractFn>
807 requires requires(const std::invoke_result_t<TMinValueExtractFn, const ItemOwned&>& a) {

810 IterValue<Item> minBy(TMinValueExtractFn minValueExtractFn) {
811 IterValue<Item> result = Iterator::next(*self());
812 if(!result.has_value()) { return {}; }
813 auto resultValue = minValueExtractFn(result.value());
814 forEach([&result, &resultValue, &minValueExtractFn](Item&& item) {
815 auto itemValue = minValueExtractFn(item);
816 if(itemValue < resultValue) {
817 result = item;
818 resultValue = itemValue;
819 }
820 });
821 return result;
