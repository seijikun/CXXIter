743 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
744 std::optional<TResult> mean(TResult sumStart = TResult()) {
745 size_t cnt = 0;
746 TResult result = fold(sumStart, [&cnt](TResult& res, Item&& item) {
747 cnt += 1;
748 res += item;
749 });
750 if(cnt > 0) {
751 if constexpr(NORM == StatisticNormalization::N) {

797 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
798 std::optional<TResult> variance() {
799 TResult sumSquare = TResult();
800 TResult sum = TResult();
801 size_t cnt = 0;
802 forEach([&sumSquare, &sum, &cnt](Item&& item) {
803 sum += item;
804 sumSquare += (item * item);
805 cnt += 1;
806 });
807 if(cnt >= 2) {
808 if constexpr(NORM == StatisticNormalization::N) {
809 TResult E1 = (sumSquare / static_cast<TCount>(cnt));
810 TResult E2 = (sum / static_cast<TCount>(cnt));
811 return E1 - (E2 * E2);
812 } else {
813 TResult E1 = (sum * sum / static_cast<TCount>(cnt));
814 return (sumSquare - E1) / static_cast<TCount>(cnt - 1);
815 }
816 }
817 return {};
818 }
