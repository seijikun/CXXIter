784 constexpr std::optional<TResult> mean(TResult sumStart = TResult()) {
785 size_t cnt = 0;
786 TResult result = fold(sumStart, [&cnt](TResult& res, Item&& item) {
787 cnt += 1;
788 res += item;
789 });
790 if(cnt > 0) {
791 if constexpr(NORM == StatisticNormalization::N) {
792 return result / static_cast<TCount>(cnt);
793 } else {
794 return result / static_cast<TCount>(cnt - 1);

837 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
838 constexpr std::optional<TResult> variance() {
839 TResult sumSquare = TResult();
840 TResult sum = TResult();
841 size_t cnt = 0;
842 forEach([&sumSquare, &sum, &cnt](Item&& item) {
843 sum += item;
844 sumSquare += (item * item);
845 cnt += 1;
846 });
847 if(cnt >= 2) {
848 if constexpr(NORM == StatisticNormalization::N) {
849 TResult E1 = (sumSquare / static_cast<TCount>(cnt));
850 TResult E2 = (sum / static_cast<TCount>(cnt));
851 return E1 - (E2 * E2);
852 } else {
853 TResult E1 = (sum * sum / static_cast<TCount>(cnt));
854 return (sumSquare - E1) / static_cast<TCount>(cnt - 1);
855 }
856 }
857 return {};
858 }
