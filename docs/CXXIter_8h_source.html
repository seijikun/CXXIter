765 std::optional<TResult> mean(TResult sumStart = TResult()) {
766 size_t cnt = 0;
767 TResult result = fold(sumStart, [&cnt](TResult& res, Item&& item) {
768 cnt += 1;
769 res += item;

818 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
819 std::optional<TResult> variance() {
820 TResult sumSquare = TResult();
821 TResult sum = TResult();
822 size_t cnt = 0;
823 forEach([&sumSquare, &sum, &cnt](Item&& item) {
824 sum += item;
825 sumSquare += (item * item);
826 cnt += 1;
827 });
828 if(cnt >= 2) {
829 if constexpr(NORM == StatisticNormalization::N) {
830 TResult E1 = (sumSquare / static_cast<TCount>(cnt));
831 TResult E2 = (sum / static_cast<TCount>(cnt));
832 return E1 - (E2 * E2);
833 } else {
834 TResult E1 = (sum * sum / static_cast<TCount>(cnt));
835 return (sumSquare - E1) / static_cast<TCount>(cnt - 1);
836 }
837 }
838 return {};
839 }
