785 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
786 constexpr std::optional<TResult> mean(TResult sumStart = TResult()) {
787 size_t cnt = 0;
788 TResult result = fold(sumStart, [&cnt](TResult& res, Item&& item) {
789 cnt += 1;
790 res += item;
791 });
792 if(cnt > 0) {
793 if constexpr(NORM == StatisticNormalization::N) {
794 return result / static_cast<TCount>(cnt);
795 } else {
796 return result / static_cast<TCount>(cnt - 1);
797 }
798 }
799 return {};

839 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
840 constexpr std::optional<TResult> variance() {
841 TResult sumSquare = TResult();
842 TResult sum = TResult();
843 size_t cnt = 0;
844 forEach([&sumSquare, &sum, &cnt](Item&& item) {
845 sum += item;
846 sumSquare += (item * item);
847 cnt += 1;
848 });
849 if(cnt >= 2) {
850 if constexpr(NORM == StatisticNormalization::N) {
851 TResult E1 = (sumSquare / static_cast<TCount>(cnt));
852 TResult E2 = (sum / static_cast<TCount>(cnt));
853 return E1 - (E2 * E2);
854 } else {
855 TResult E1 = (sum * sum / static_cast<TCount>(cnt));
856 return (sumSquare - E1) / static_cast<TCount>(cnt - 1);
857 }
858 }
859 return {};
860 }
