708 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
709 std::optional<TResult> mean(TResult sumStart = TResult()) {
710 size_t cnt = 0;
711 TResult result = fold(sumStart, [&cnt](TResult& res, Item&& item) {
712 cnt += 1;
713 res += item;
714 });
715 if(cnt > 0) {
716 if constexpr(NORM == StatisticNormalization::N) {
717 return result / static_cast<TCount>(cnt);
718 } else {
719 return result / static_cast<TCount>(cnt - 1);
720 }
721 }
722 return {};
723 }

762 template<StatisticNormalization NORM = StatisticNormalization::N, typename TResult = ItemOwned, typename TCount = ItemOwned>
763 std::optional<TResult> variance() {
764 TResult sumSquare = TResult();
765 TResult sum = TResult();
766 size_t cnt = 0;
767 forEach([&sumSquare, &sum, &cnt](Item&& item) {
768 sum += item;
769 sumSquare += (item * item);
770 cnt += 1;
771 });
772 if(cnt >= 2) {
773 if constexpr(NORM == StatisticNormalization::N) {
774 TResult E1 = (sumSquare / static_cast<TCount>(cnt));
775 TResult E2 = (sum / static_cast<TCount>(cnt));
776 return E1 - (E2 * E2);
777 } else {
778 TResult E1 = (sum * sum / static_cast<TCount>(cnt));
779 return (sumSquare - E1) / static_cast<TCount>(cnt - 1);
780 }
781 }
782 return {};
783 }
