39
40 static inline IterValue<Item> next(Self& self) {
41 if(!self.sortCache.has_value()) {
42 // drain input iterator into sortCache
43 std::vector<OwnedInputItem> sortCache;
44 while(true) {
45 auto item = ChainInputIterator::next(self.input);
46 if(!item.has_value()) [[unlikely]] { break; }
47 sortCache.push_back(std::forward<InputItem>( item.value() ));
48 }
49 // sort the cache
50 if constexpr(STABLE) {
51 std::stable_sort(sortCache.begin(), sortCache.end(), self.compareFn);
52 } else {
53 std::sort(sortCache.begin(), sortCache.end(), self.compareFn);
54 }
55 self.sortCache.emplace(std::move(sortCache));
56 }
57
58 using SortCacheIterator = IteratorTrait<typename Self::SortCache>;
59 typename Self::SortCache& sortedItems = self.sortCache.value();
60 return SortCacheIterator::next(sortedItems);
61 }
62 static inline SizeHint sizeHint(const Self& self) { return ChainInputIterator::sizeHint(self.input); }
63 };
