590// FLATMAP
591// ################################################################################################
593template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
594requires (!std::is_reference_v<TItemContainer>)
595class FlatMap : public IterApi<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
596 friend struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>>;
597private:
598 TChainInput input;
599 std::optional<SrcMov<TItemContainer>> current;
600 TFlatMapFn mapFn;
601public:
602 FlatMap(TChainInput&& input, TFlatMapFn mapFn) : input(std::move(input)), mapFn(mapFn) {}
603};
604// ------------------------------------------------------------------------------------------------
606template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
607struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
608 using NestedChainIterator = IteratorTrait<SrcMov<TItemContainer>>;
609 using ChainInputIterator = IteratorTrait<TChainInput>;
610 using InputItem = typename ChainInputIterator::Item;
611 // CXXIter Interface
612 using Self = FlatMap<TChainInput, TFlatMapFn, TItemContainer>;
613 using Item = typename TItemContainer::value_type;
614
615 static inline IterValue<Item> next(Self& self) {
616 while(true) {
617 if(!self.current) { // pull new container from the outer iterator
618 auto item = ChainInputIterator::next(self.input);
619 if(!item.hasValue()) { return {}; } // end of iteration
620 self.current = SrcMov(std::move(
621 self.mapFn(std::forward<InputItem>( item.value() ))
622 ));
623 }
624
625 // if the outer iterator yielded a container, take from it until we reach the end
626 auto item = NestedChainIterator::next(*self.current);
627 if(item.hasValue()) { // inner yielded a usable item
628 return item.value();
629 } else {
630 self.current.reset(); // inner container ended, unset current cache
631 }
632 }
633 }
634};

745// GROUP BY
746// ################################################################################################
748template<typename TChainInput, typename TGroupIdentifierFn, typename TGroupIdent>
749class GroupBy : public IterApi<GroupBy<TChainInput, TGroupIdentifierFn, TGroupIdent>> {
750 friend struct IteratorTrait<GroupBy<TChainInput, TGroupIdentifierFn, TGroupIdent>>;
751private:
752 using OwnedInputItem = typename TChainInput::ItemOwned;
753 using GroupCache = SrcMov<std::unordered_map<TGroupIdent, std::vector<OwnedInputItem>>>;
754
755 TChainInput input;
756 TGroupIdentifierFn groupIdentFn;
757 std::optional<GroupCache> groupCache;
758public:
759 GroupBy(TChainInput&& input, TGroupIdentifierFn groupIdentFn) : input(std::move(input)), groupIdentFn(groupIdentFn) {}
760};
761// ------------------------------------------------------------------------------------------------
763template<typename TChainInput, typename TGroupIdentifierFn, typename TGroupIdent>
764struct IteratorTrait<GroupBy<TChainInput, TGroupIdentifierFn, TGroupIdent>> {
765 using ChainInputIterator = IteratorTrait<TChainInput>;
766 using OwnedInputItem = typename TChainInput::ItemOwned;
767 // CXXIter Interface
768 using Self = GroupBy<TChainInput, TGroupIdentifierFn, TGroupIdent>;
769 using Item = std::pair<const TGroupIdent, std::vector<OwnedInputItem>>;
770
771 static inline IterValue<Item> next(Self& self) {
772 // we have to drain the input in order to be able to calculate the groups
773 // so we do that on the first invocation, and then yield from the calculated result.
774 if(!self.groupCache.has_value()) [[unlikely]] {
775 std::unordered_map<TGroupIdent, std::vector<OwnedInputItem>> groupCache;
776 while(true) {
777 auto item = ChainInputIterator::next(self.input);
778 if(!item.hasValue()) { break; } // group cache building complete
779 TGroupIdent itemGroup = self.groupIdentFn(item.value());
780 if(groupCache.contains(itemGroup)) {
781 groupCache[itemGroup].push_back(item.value());
782 } else {
783 groupCache[itemGroup] = { item.value() };
784 }
785 }
786 self.groupCache.emplace(std::move(groupCache));
787 }
788
789 using GroupCacheIterator = IteratorTrait<typename Self::GroupCache>;
790 typename Self::GroupCache& groupedItems = self.groupCache.value();
791 return GroupCacheIterator::next(groupedItems);
792 }
793};

803 friend struct IteratorTrait<Sorter<TChainInput, TCompareFn, STABLE>>;
804private:
805 using OwnedInputItem = typename TChainInput::ItemOwned;
806 using SortCache = SrcMov<std::vector<OwnedInputItem>>;
807
808 TChainInput input;
809 TCompareFn compareFn;
810 std::optional<SortCache> sortCache;
811public:
812 Sorter(TChainInput&& input, TCompareFn compareFn) : input(std::move(input)), compareFn(compareFn) {}
813};
814// ------------------------------------------------------------------------------------------------
816template<typename TChainInput, typename TCompareFn, bool STABLE>
817struct IteratorTrait<Sorter<TChainInput, TCompareFn, STABLE>> {
818 using ChainInputIterator = IteratorTrait<TChainInput>;
819 using InputItem = typename TChainInput::Item;
820 using OwnedInputItem = typename TChainInput::ItemOwned;
821 // CXXIter Interface
822 using Self = Sorter<TChainInput, TCompareFn, STABLE>;
823 using Item = OwnedInputItem;
824
825 static inline IterValue<Item> next(Self& self) {
826 if(!self.sortCache.has_value()) {
827 // drain input iterator into sortCache
828 std::vector<OwnedInputItem> sortCache;
829 while(true) {
830 auto item = ChainInputIterator::next(self.input);
831 if(!item.hasValue()) { break; }
832 sortCache.push_back(std::forward<InputItem>( item.value() ));
833 }
834 // sort the cache
835 if constexpr(STABLE) {
836 std::stable_sort(sortCache.begin(), sortCache.end(), self.compareFn);
837 } else {
838 std::sort(sortCache.begin(), sortCache.end(), self.compareFn);
839 }
840 self.sortCache.emplace(std::move(sortCache));
841 }
842
843 using SortCacheIterator = IteratorTrait<typename Self::SortCache>;
844 typename Self::SortCache& sortedItems = self.sortCache.value();
845 return SortCacheIterator::next(sortedItems);
846 }
847};
