122struct SizeHint {
123 constexpr static size_t INFINITE = std::numeric_limits<size_t>::max();
124
125 size_t lowerBound;
126 std::optional<size_t> upperBound;
127
128 size_t expectedResultSize(size_t min = 0) const { return std::min(min, upperBound.value_or(lowerBound)); }
129
130 SizeHint(size_t lowerBound = 0, std::optional<size_t> upperBound = {}) : lowerBound(lowerBound), upperBound(upperBound) {}
131
132 static std::optional<size_t> upperBoundMax(std::optional<size_t> upperBound1, std::optional<size_t> upperBound2) {
133 if(!upperBound1.has_value() || !upperBound2.has_value()) { return {}; } // no upperbound is like Infinity -> higher
134 return std::max(upperBound1.value(), upperBound2.value());
135 }
136 static std::optional<size_t> upperBoundMin(std::optional<size_t> upperBound1, std::optional<size_t> upperBound2) {
137 if(!upperBound1.has_value()) { return upperBound2; }
138 if(!upperBound2.has_value()) { return upperBound1; }
139 return std::min(upperBound1.value(), upperBound2.value());
140 }
141 void subtract(size_t cnt) {
142 lowerBound = (lowerBound > cnt) ? (lowerBound - cnt) : 0;
143 if(upperBound) {
144 upperBound = (upperBound.value() > cnt) ? (upperBound.value() - cnt) : 0;
145 }
146 }
147 void add(const SizeHint& o) {
148 lowerBound += o.lowerBound;
149 if(upperBound.has_value() && o.upperBound.has_value()) {
150 upperBound = upperBound.value() + o.upperBound.value();
151 } else {
152 upperBound = {};
153 }
154 }
155};

747// ################################################################################################
749template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
750requires (!std::is_reference_v<TItemContainer>)
751class FlatMap : public IterApi<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
752 friend struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>>;
753private:
754 TChainInput input;
755 std::optional<SrcMov<TItemContainer>> current;
756 TFlatMapFn mapFn;
757public:
758 FlatMap(TChainInput&& input, TFlatMapFn mapFn) : input(std::move(input)), mapFn(mapFn) {}
759};
760// ------------------------------------------------------------------------------------------------
762template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
763struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
764 using NestedChainIterator = IteratorTrait<SrcMov<TItemContainer>>;
765 using ChainInputIterator = IteratorTrait<TChainInput>;
766 using InputItem = typename ChainInputIterator::Item;
767 // CXXIter Interface
768 using Self = FlatMap<TChainInput, TFlatMapFn, TItemContainer>;
769 using Item = typename TItemContainer::value_type;
770
771 static inline IterValue<Item> next(Self& self) {
772 while(true) {
773 if(!self.current) { // pull new container from the outer iterator
774 auto item = ChainInputIterator::next(self.input);
775 if(!item.has_value()) { return {}; } // end of iteration
776 self.current = SrcMov(std::move(
777 self.mapFn(std::forward<InputItem>( item.value() ))
778 ));
779 }
780
781 // if the outer iterator yielded a container, take from it until we reach the end
782 auto item = NestedChainIterator::next(*self.current);
783 if(item.has_value()) { // inner yielded a usable item
784 return item.value();
785 } else {
786 self.current.reset(); // inner container ended, unset current cache
787 }
788 }
789 }
790 static inline SizeHint sizeHint(const Self&) { return SizeHint(); }
791};

1055 });
1056 self.batchElementIdx = 0;
1057 }
1058 return std::move(self.currentBatch[self.batchElementIdx++]);

1061 size_t lowerBoundMin = std::numeric_limits<size_t>::max();

1066 if(lowerBoundMin > tmp.lowerBound) {
1067 minIdx = idx;
1068 lowerBoundMin = tmp.lowerBound;
1069 }
1070 upperBoundMin = SizeHint::upperBoundMin(upperBoundMin, tmp.upperBound);
1071 return true;
1072 });
1073 // smallest input iterator defines base length. The amount of elements then is that
1074 // length, multiplied by the amount of input iterators + the index of the shortest
1075 if(upperBoundMin.has_value()) { upperBoundMin.value() = upperBoundMin.value() * Self::BATCH_SIZE + minIdx; }

1077 lowerBoundMin * Self::BATCH_SIZE + minIdx,
1078 upperBoundMin
1079 );
1080 return result;
