35 inline const TValue& value() const { return inner.value(); }
36 inline TValue& value() { return inner.value(); }
37 inline const TValue& value(TValue&& def) const { return inner.value_or(def); }
38 inline TValue& value(TValue&& def) { return inner.value_or(def); }
39
40 bool hasValue() const { return inner.has_value(); }
41
42 IterValue<TValue>& operator=(TValue&& o) {
43 inner = std::forward<TValue>(o);
44 return *this;
45 }
46
47 template<typename TOutValue, std::invocable<TValue&&> TMapFn>
48 IterValue<TOutValue> map(TMapFn mapFn) {
49 if(!hasValue()) { return {}; }
50 return mapFn(std::forward<TValue>(value()));
51 }

104struct SizeHint {
105 size_t lowerBound;
106 std::optional<size_t> upperBound;
107
108 size_t expectedResultSize(size_t min = 0) const { return std::min(min, upperBound.value_or(lowerBound)); }
109
110 SizeHint(size_t lowerBound = 0, std::optional<size_t> upperBound = {}) : lowerBound(lowerBound), upperBound(upperBound) {}
111
112 static std::optional<size_t> upperBoundMax(std::optional<size_t> upperBound1, std::optional<size_t> upperBound2) {
113 if(!upperBound1.has_value() || !upperBound2.has_value()) { return {}; } // no upperbound is like Infinity -> higher
114 return std::max(upperBound1.value(), upperBound2.value());
115 }
116 static std::optional<size_t> upperBoundMin(std::optional<size_t> upperBound1, std::optional<size_t> upperBound2) {
117 if(!upperBound1.has_value()) { return upperBound2; }
118 if(!upperBound2.has_value()) { return upperBound1; }
119 return std::min(upperBound1.value(), upperBound2.value());
120 }
121 void subtract(size_t cnt) {
122 lowerBound = (lowerBound > cnt) ? (lowerBound - cnt) : 0;
123 if(upperBound) {
124 upperBound = (upperBound.value() > cnt) ? (upperBound.value() - cnt) : 0;
125 }
126 }
127};

269 static inline bool hasNext(const TContainer& container, ConstIteratorState& iter) { return (iter != container.end()); }

278 static inline Item& next([[maybe_unused]] TContainer& container, IteratorState& iter) { return (*iter++); }

497class Range : public IterApi<Range<TValue>> {
498 friend struct IteratorTrait<Range<TValue>>;
499private:
500 TValue current;
501 TValue from;

510struct IteratorTrait<Range<TValue>> {
511 // CXXIter Interface
512 using Self = Range<TValue>;
513 using Item = TValue;
514
515 static inline IterValue<Item> next(Self& self) {
516 if(self.current > self.to) { return {}; }
517 TValue current = self.current;
518 self.current += self.step;
519 return current;
520 }
521 static inline SizeHint sizeHint(const Self& self) {
522 size_t cnt = static_cast<size_t>((self.to - self.from) / self.step) + 1;
523 return SizeHint(cnt, cnt);
524 }

669// ################################################################################################
670// FLATMAP
671// ################################################################################################
673template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
674requires (!std::is_reference_v<TItemContainer>)
675class FlatMap : public IterApi<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
676 friend struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>>;
677private:
678 TChainInput input;
679 std::optional<SrcMov<TItemContainer>> current;
680 TFlatMapFn mapFn;
681public:
682 FlatMap(TChainInput&& input, TFlatMapFn mapFn) : input(std::move(input)), mapFn(mapFn) {}
683};
684// ------------------------------------------------------------------------------------------------
686template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
687struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
688 using NestedChainIterator = IteratorTrait<SrcMov<TItemContainer>>;
689 using ChainInputIterator = IteratorTrait<TChainInput>;
690 using InputItem = typename ChainInputIterator::Item;
691 // CXXIter Interface
692 using Self = FlatMap<TChainInput, TFlatMapFn, TItemContainer>;
693 using Item = typename TItemContainer::value_type;
694
695 static inline IterValue<Item> next(Self& self) {
696 while(true) {
697 if(!self.current) { // pull new container from the outer iterator
698 auto item = ChainInputIterator::next(self.input);
699 if(!item.hasValue()) { return {}; } // end of iteration
700 self.current = SrcMov(std::move(
701 self.mapFn(std::forward<InputItem>( item.value() ))
702 ));
703 }
704
705 // if the outer iterator yielded a container, take from it until we reach the end
706 auto item = NestedChainIterator::next(*self.current);
707 if(item.hasValue()) { // inner yielded a usable item
708 return item.value();
709 } else {
710 self.current.reset(); // inner container ended, unset current cache
711 }
712 }
713 }
714 static inline SizeHint sizeHint(const Self&) { return SizeHint(); }
715};

906
907 TChainInput input;
908 TGroupIdentifierFn groupIdentFn;

910public:

912};

915template<typename TChainInput, typename TGroupIdentifierFn, typename TGroupIdent>
916struct IteratorTrait<GroupBy<TChainInput, TGroupIdentifierFn, TGroupIdent>> {
917 using ChainInputIterator = IteratorTrait<TChainInput>;
918 using OwnedInputItem = typename TChainInput::ItemOwned;
919 // CXXIter Interface
920 using Self = GroupBy<TChainInput, TGroupIdentifierFn, TGroupIdent>;
921 using Item = std::pair<const TGroupIdent, std::vector<OwnedInputItem>>;
922
923 static inline IterValue<Item> next(Self& self) {
924 // we have to drain the input in order to be able to calculate the groups
925 // so we do that on the first invocation, and then yield from the calculated result.
926 if(!self.groupCache.has_value()) [[unlikely]] {
927 std::unordered_map<TGroupIdent, std::vector<OwnedInputItem>> groupCache;
928 while(true) {
929 auto item = ChainInputIterator::next(self.input);
930 if(!item.hasValue()) { break; } // group cache building complete
931 TGroupIdent itemGroup = self.groupIdentFn(item.value());
932 if(groupCache.contains(itemGroup)) {
933 groupCache[itemGroup].push_back(item.value());
934 } else {
935 groupCache[itemGroup] = { item.value() };
936 }
937 }
938 self.groupCache.emplace(std::move(groupCache));
939 }
940
941 using GroupCacheIterator = IteratorTrait<typename Self::GroupCache>;
942 typename Self::GroupCache& groupedItems = self.groupCache.value();
943 return GroupCacheIterator::next(groupedItems);
944 }
945 static inline SizeHint sizeHint(const Self& self) {
946 SizeHint input = ChainInputIterator::sizeHint(self.input);
947 return SizeHint(1, input.upperBound);
948 }

959 friend struct IteratorTrait<Sorter<TChainInput, TCompareFn, STABLE>>;
960private:
961 using OwnedInputItem = typename TChainInput::ItemOwned;
962 using SortCache = SrcMov<std::vector<OwnedInputItem>>;
963
964 TChainInput input;
965 TCompareFn compareFn;
966 std::optional<SortCache> sortCache;
967public:
968 Sorter(TChainInput&& input, TCompareFn compareFn) : input(std::move(input)), compareFn(compareFn) {}
969};
970// ------------------------------------------------------------------------------------------------
972template<typename TChainInput, typename TCompareFn, bool STABLE>
973struct IteratorTrait<Sorter<TChainInput, TCompareFn, STABLE>> {
974 using ChainInputIterator = IteratorTrait<TChainInput>;
975 using InputItem = typename TChainInput::Item;
976 using OwnedInputItem = typename TChainInput::ItemOwned;
977 // CXXIter Interface
978 using Self = Sorter<TChainInput, TCompareFn, STABLE>;
979 using Item = OwnedInputItem;
980
981 static inline IterValue<Item> next(Self& self) {
982 if(!self.sortCache.has_value()) {
983 // drain input iterator into sortCache
984 std::vector<OwnedInputItem> sortCache;
985 while(true) {
986 auto item = ChainInputIterator::next(self.input);
987 if(!item.hasValue()) { break; }
988 sortCache.push_back(std::forward<InputItem>( item.value() ));
989 }
990 // sort the cache
991 if constexpr(STABLE) {
992 std::stable_sort(sortCache.begin(), sortCache.end(), self.compareFn);
993 } else {
994 std::sort(sortCache.begin(), sortCache.end(), self.compareFn);
995 }
996 self.sortCache.emplace(std::move(sortCache));
997 }
998
999 using SortCacheIterator = IteratorTrait<typename Self::SortCache>;
1000 typename Self::SortCache& sortedItems = self.sortCache.value();
1001 return SortCacheIterator::next(sortedItems);
1002 }
1003 static inline SizeHint sizeHint(const Self& self) { return ChainInputIterator::sizeHint(self.input); }
1004};
