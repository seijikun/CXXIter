591// FLATMAP
592// ################################################################################################
594template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
595requires (!std::is_reference_v<TItemContainer>)
596class FlatMap : public IterApi<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
597 friend struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>>;
598private:
599 TChainInput input;
600 std::optional<SrcMov<TItemContainer>> current;
601 TFlatMapFn mapFn;
602public:
603 FlatMap(TChainInput&& input, TFlatMapFn mapFn) : input(std::move(input)), mapFn(mapFn) {}
604};
605// ------------------------------------------------------------------------------------------------
607template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
608struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
609 using NestedChainIterator = IteratorTrait<SrcMov<TItemContainer>>;
610 using ChainInputIterator = IteratorTrait<TChainInput>;
611 using InputItem = typename ChainInputIterator::Item;
612 // CXXIter Interface
613 using Self = FlatMap<TChainInput, TFlatMapFn, TItemContainer>;
614 using Item = typename TItemContainer::value_type;
615
616 static inline IterValue<Item> next(Self& self) {
617 while(true) {
618 if(!self.current) { // pull new container from the outer iterator
619 auto item = ChainInputIterator::next(self.input);
620 if(!item.hasValue()) { return {}; } // end of iteration
621 self.current = SrcMov(std::move(
622 self.mapFn(std::forward<InputItem>( item.value() ))
623 ));
624 }
625
626 // if the outer iterator yielded a container, take from it until we reach the end
627 auto item = NestedChainIterator::next(*self.current);
628 if(item.hasValue()) { // inner yielded a usable item
629 return item.value();
630 } else {
631 self.current.reset(); // inner container ended, unset current cache
632 }
633 }
634 }
635};

747// ################################################################################################
749template<typename TChainInput, typename TGroupIdentifierFn, typename TGroupIdent>
750class GroupBy : public IterApi<GroupBy<TChainInput, TGroupIdentifierFn, TGroupIdent>> {
751 friend struct IteratorTrait<GroupBy<TChainInput, TGroupIdentifierFn, TGroupIdent>>;
752private:
753 using OwnedInputItem = typename TChainInput::ItemOwned;
754 using GroupCache = SrcMov<std::unordered_map<TGroupIdent, std::vector<OwnedInputItem>>>;
755
756 TChainInput input;
757 TGroupIdentifierFn groupIdentFn;
758 std::optional<GroupCache> groupCache;
759public:
760 GroupBy(TChainInput&& input, TGroupIdentifierFn groupIdentFn) : input(std::move(input)), groupIdentFn(groupIdentFn) {}
761};
762// ------------------------------------------------------------------------------------------------
764template<typename TChainInput, typename TGroupIdentifierFn, typename TGroupIdent>
765struct IteratorTrait<GroupBy<TChainInput, TGroupIdentifierFn, TGroupIdent>> {
766 using ChainInputIterator = IteratorTrait<TChainInput>;
767 using OwnedInputItem = typename TChainInput::ItemOwned;
768 // CXXIter Interface
769 using Self = GroupBy<TChainInput, TGroupIdentifierFn, TGroupIdent>;
770 using Item = std::pair<const TGroupIdent, std::vector<OwnedInputItem>>;
771
772 static inline IterValue<Item> next(Self& self) {
773 // we have to drain the input in order to be able to calculate the groups
774 // so we do that on the first invocation, and then yield from the calculated result.
775 if(!self.groupCache.has_value()) [[unlikely]] {
776 std::unordered_map<TGroupIdent, std::vector<OwnedInputItem>> groupCache;
777 while(true) {
778 auto item = ChainInputIterator::next(self.input);
779 if(!item.hasValue()) { break; } // group cache building complete
780 TGroupIdent itemGroup = self.groupIdentFn(item.value());
781 if(groupCache.contains(itemGroup)) {
782 groupCache[itemGroup].push_back(item.value());
783 } else {
784 groupCache[itemGroup] = { item.value() };
785 }
786 }
787 self.groupCache.emplace(std::move(groupCache));
788 }
789
790 using GroupCacheIterator = IteratorTrait<typename Self::GroupCache>;
791 typename Self::GroupCache& groupedItems = self.groupCache.value();
792 return GroupCacheIterator::next(groupedItems);
793 }
794};

804 friend struct IteratorTrait<Sorter<TChainInput, TCompareFn, STABLE>>;
805private:
806 using OwnedInputItem = typename TChainInput::ItemOwned;
807 using SortCache = SrcMov<std::vector<OwnedInputItem>>;
808
809 TChainInput input;
810 TCompareFn compareFn;
811 std::optional<SortCache> sortCache;
812public:
813 Sorter(TChainInput&& input, TCompareFn compareFn) : input(std::move(input)), compareFn(compareFn) {}
814};

817template<typename TChainInput, typename TCompareFn, bool STABLE>
818struct IteratorTrait<Sorter<TChainInput, TCompareFn, STABLE>> {
819 using ChainInputIterator = IteratorTrait<TChainInput>;
820 using InputItem = typename TChainInput::Item;
821 using OwnedInputItem = typename TChainInput::ItemOwned;
822 // CXXIter Interface
823 using Self = Sorter<TChainInput, TCompareFn, STABLE>;
824 using Item = OwnedInputItem;
825
826 static inline IterValue<Item> next(Self& self) {
827 if(!self.sortCache.has_value()) {
828 // drain input iterator into sortCache
829 std::vector<OwnedInputItem> sortCache;
830 while(true) {
831 auto item = ChainInputIterator::next(self.input);
832 if(!item.hasValue()) { break; }
833 sortCache.push_back(std::forward<InputItem>( item.value() ));
834 }
835 // sort the cache
836 if constexpr(STABLE) {
837 std::stable_sort(sortCache.begin(), sortCache.end(), self.compareFn);
838 } else {
839 std::sort(sortCache.begin(), sortCache.end(), self.compareFn);
840 }
841 self.sortCache.emplace(std::move(sortCache));
842 }
843
844 using SortCacheIterator = IteratorTrait<typename Self::SortCache>;
845 typename Self::SortCache& sortedItems = self.sortCache.value();
846 return SortCacheIterator::next(sortedItems);
847 }
848};
