608 template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
609 struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
610  using NestedChainIterator = IteratorTrait<SrcMov<TItemContainer>>;
611  using ChainInputIterator = IteratorTrait<TChainInput>;
612  using InputItem = typename ChainInputIterator::Item;
613  // CXXIter Interface
614  using Self = FlatMap<TChainInput, TFlatMapFn, TItemContainer>;
615  using Item = typename TItemContainer::value_type;
616 
617  static inline IterValue<Item> next(Self& self) {
618  while(true) {
619  if(!self.current) { // pull new container from the outer iterator
620  auto item = ChainInputIterator::next(self.input);
621  if(!item.hasValue()) { return {}; } // end of iteration
622  self.current = SrcMov(std::move(
623  self.mapFn(std::forward<InputItem>( item.value() ))
624  ));
625  }
626 
627  // if the outer iterator yielded a container, take from it until we reach the end
628  auto item = NestedChainIterator::next(*self.current);
629  if(item.hasValue()) { // inner yielded a usable item
630  return item.value();
631  } else {
632  self.current.reset(); // inner container ended, unset current cache
633  }
634  }
635  }
