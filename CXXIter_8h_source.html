122struct SizeHint {
123 constexpr static size_t INFINITE = std::numeric_limits<size_t>::max();
124
125 size_t lowerBound;
126 std::optional<size_t> upperBound;
127
128 size_t expectedResultSize(size_t min = 0) const { return std::min(min, upperBound.value_or(lowerBound)); }
129
130 SizeHint(size_t lowerBound = 0, std::optional<size_t> upperBound = {}) : lowerBound(lowerBound), upperBound(upperBound) {}
131
132 static std::optional<size_t> upperBoundMax(std::optional<size_t> upperBound1, std::optional<size_t> upperBound2) {
133 if(!upperBound1.has_value() || !upperBound2.has_value()) { return {}; } // no upperbound is like Infinity -> higher
134 return std::max(upperBound1.value(), upperBound2.value());
135 }
136 static std::optional<size_t> upperBoundMin(std::optional<size_t> upperBound1, std::optional<size_t> upperBound2) {
137 if(!upperBound1.has_value()) { return upperBound2; }
138 if(!upperBound2.has_value()) { return upperBound1; }
139 return std::min(upperBound1.value(), upperBound2.value());
140 }
141 void subtract(size_t cnt) {
142 lowerBound = (lowerBound > cnt) ? (lowerBound - cnt) : 0;
143 if(upperBound) {
144 upperBound = (upperBound.value() > cnt) ? (upperBound.value() - cnt) : 0;
145 }
146 }
147 void add(const SizeHint& o) {
148 lowerBound += o.lowerBound;
149 if(upperBound.has_value() && o.upperBound.has_value()) {
150 upperBound = upperBound.value() + o.upperBound.value();
151 } else {
152 upperBound = {};
153 }
154 }
155};

770template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
771requires (!std::is_reference_v<TItemContainer>)
772class FlatMap : public IterApi<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
773 friend struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>>;
774private:
775 TChainInput input;
776 std::optional<SrcMov<TItemContainer>> current;
777 TFlatMapFn mapFn;
778public:
779 FlatMap(TChainInput&& input, TFlatMapFn mapFn) : input(std::move(input)), mapFn(mapFn) {}
780};
781// ------------------------------------------------------------------------------------------------
783template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
784struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
785 using NestedChainIterator = IteratorTrait<SrcMov<TItemContainer>>;
786 using ChainInputIterator = IteratorTrait<TChainInput>;
787 using InputItem = typename ChainInputIterator::Item;
788 // CXXIter Interface
789 using Self = FlatMap<TChainInput, TFlatMapFn, TItemContainer>;
790 using Item = typename TItemContainer::value_type;
791
792 static inline IterValue<Item> next(Self& self) {
793 while(true) {
794 if(!self.current) { // pull new container from the outer iterator
795 auto item = ChainInputIterator::next(self.input);
796 if(!item.has_value()) [[unlikely]] { return {}; } // end of iteration
797 self.current = SrcMov(std::move(
798 self.mapFn(std::forward<InputItem>( item.value() ))
799 ));
800 }
801
802 // if the outer iterator yielded a container, take from it until we reach the end
803 auto item = NestedChainIterator::next(*self.current);
804 if(item.has_value()) [[likely]] { // inner yielded a usable item
805 return item.value();
806 } else [[unlikely]] {
807 self.current.reset(); // inner container ended, unset current cache
808 }
809 }
810 }
811 static inline SizeHint sizeHint(const Self&) { return SizeHint(); }
812};
