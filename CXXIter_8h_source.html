109struct SizeHint {
110 size_t lowerBound;
111 std::optional<size_t> upperBound;
112
113 size_t expectedResultSize(size_t min = 0) const { return std::min(min, upperBound.value_or(lowerBound)); }
114
115 SizeHint(size_t lowerBound = 0, std::optional<size_t> upperBound = {}) : lowerBound(lowerBound), upperBound(upperBound) {}
116
117 static std::optional<size_t> upperBoundMax(std::optional<size_t> upperBound1, std::optional<size_t> upperBound2) {
118 if(!upperBound1.has_value() || !upperBound2.has_value()) { return {}; } // no upperbound is like Infinity -> higher
119 return std::max(upperBound1.value(), upperBound2.value());
120 }
121 static std::optional<size_t> upperBoundMin(std::optional<size_t> upperBound1, std::optional<size_t> upperBound2) {
122 if(!upperBound1.has_value()) { return upperBound2; }
123 if(!upperBound2.has_value()) { return upperBound1; }
124 return std::min(upperBound1.value(), upperBound2.value());
125 }
126 void subtract(size_t cnt) {
127 lowerBound = (lowerBound > cnt) ? (lowerBound - cnt) : 0;
128 if(upperBound) {
129 upperBound = (upperBound.value() > cnt) ? (upperBound.value() - cnt) : 0;
130 }
131 }
132 void add(const SizeHint& o) {
133 lowerBound += o.lowerBound;
134 if(upperBound.has_value() && o.upperBound.has_value()) {
135 upperBound = upperBound.value() + o.upperBound.value();
136 } else {
137 upperBound = {};
138 }
139 }
140};

265 static inline ConstIteratorState initIterator(const TContainer& container) { return container.begin(); }

282 static inline bool hasNext(const TContainer& container, ConstIteratorState& iter) { return (iter != container.end()); }

683// FLATMAP
684// ################################################################################################
686template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
687requires (!std::is_reference_v<TItemContainer>)
688class FlatMap : public IterApi<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
689 friend struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>>;
690private:
691 TChainInput input;
692 std::optional<SrcMov<TItemContainer>> current;
693 TFlatMapFn mapFn;
694public:
695 FlatMap(TChainInput&& input, TFlatMapFn mapFn) : input(std::move(input)), mapFn(mapFn) {}
696};
697// ------------------------------------------------------------------------------------------------
699template<typename TChainInput, typename TFlatMapFn, typename TItemContainer>
700struct IteratorTrait<FlatMap<TChainInput, TFlatMapFn, TItemContainer>> {
701 using NestedChainIterator = IteratorTrait<SrcMov<TItemContainer>>;
702 using ChainInputIterator = IteratorTrait<TChainInput>;
703 using InputItem = typename ChainInputIterator::Item;
704 // CXXIter Interface
705 using Self = FlatMap<TChainInput, TFlatMapFn, TItemContainer>;
706 using Item = typename TItemContainer::value_type;
707
708 static inline IterValue<Item> next(Self& self) {
709 while(true) {
710 if(!self.current) { // pull new container from the outer iterator
711 auto item = ChainInputIterator::next(self.input);
712 if(!item.hasValue()) { return {}; } // end of iteration
713 self.current = SrcMov(std::move(
714 self.mapFn(std::forward<InputItem>( item.value() ))
715 ));
716 }
717
718 // if the outer iterator yielded a container, take from it until we reach the end
719 auto item = NestedChainIterator::next(*self.current);
720 if(item.hasValue()) { // inner yielded a usable item
721 return item.value();
722 } else {
723 self.current.reset(); // inner container ended, unset current cache
724 }
725 }
726 }
727 static inline SizeHint sizeHint(const Self&) { return SizeHint(); }
728};

985 static inline IterValue<Item> next(Self& self) {
986 // we have to drain the input in order to be able to calculate the groups
987 // so we do that on the first invocation, and then yield from the calculated result.
988 if(!self.groupCache.has_value()) [[unlikely]] {
989 std::unordered_map<TGroupIdent, std::vector<OwnedInputItem>> groupCache;

992 if(!item.hasValue()) { break; } // group cache building complete

998 }
999 }
1000 self.groupCache.emplace(std::move(groupCache));
1001 }

1004 typename Self::GroupCache& groupedItems = self.groupCache.value();

1006 }
